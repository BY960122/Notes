# 全局环境变量
```sh
vim ~/.bash_profile

# Get the aliases and functions
if [ -f ~/.bashrc ]; then
        . ~/.bashrc
fi

# User specific environment and startup programs

PATH=$PATH:$HOME/.local/bin:$HOME/bin

export JAVA_HOME=/opt/software/jdk1.8.0_291
export SCALA_HOME=/opt/software/scala-2.12.11
export MYSQL_HOME=/opt/software/mysql-8.0.21
export PYTHON_HOME=/usr/local/python3
export HADOOP_HOME=/opt/software/hadoop-3.2.1
export ZOOKEEPER_HOME=/opt/software/zookeeper-3.6.1
export HIVE_HOME=/opt/software/apache-hive-3.1.1
export HBASE_HOME=/opt/software/hbase-2.4.1
export SPARK_HOME=/opt/software/spark-3.1.1-bin-hadoop3.2
export KAFKA_HOME=/opt/software/kafka_2.12-2.8.1
export ELASTICSEARCH_HOME=/opt/software/elasticsearch-7.13.1
export STORM_HOME=/opt/software/apache-storm-2.2.0
export FLINK_HOME=/opt/software/flink-1.13.1
export FLUME_HOME=/opt/software/apache-flume-1.9.0
export SQOOP_HOME=/opt/software/sqoop-1.4.7.bin__hadoop-2.6.0

# hadoop 3.0 才需要配置
export HDFS_NAMENODE_USER=root
export HDFS_DATANODE_USER=root
export HDFS_JOURNALNODE_USER=root
export HDFS_SECONDARYNAMENODE_USER=root
export HDFS_ZKFC_USER=root
export YARN_RESOURCEMANAGER_USER=root
export YARN_NODEMANAGER_USER=root

PATH=$PATH:$HOME/bin:$JAVA_HOME/bin:$SCALA_HOME/bin:$MYSQL_HOME/bin:$PYTHON_HOME/bin:$HADOOP_HOME/bin:$HADOOP_HOME/sbin:$ZOOKEEPER_HOME/bin:$HIVE_HOME/bin:$HIVE_HOME/lib:$HBASE_HOME/bin:$HBASE_HOME/lib:$SPARK_HOME/bin:$KAFKA_HOME/bin:$ELASTICSEARCH_HOME/bin:$STORM_HOME/bin:$FLINK_HOME/bin:$FLUME_HOME/bin:$SQOOP_HOME/bin

export PATH
export HADOOP_CLASSPATH=`hadoop classpath`

source ~/.bash_profile
```
